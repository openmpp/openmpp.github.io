91namespace std {
92 template <>
93 struct hash<omc::position>
94 {
95 size_t operator()(const omc::position& k) const
96 {
97 // Comments are uniquely determined by first position in location
98 return (
99 (hash<unsigned int>()(k.column))
100 ^ (hash<unsigned int>()(k.line) << 7) // assume max 128 columns
101 ^ (hash<unsigned int>()(*k.filename->cbegin()) << 10) // assume max 1024 lines
102 );
103 }
104 };
105}

120// TODO: Explain better the use of double indirection to persist of Symbols and allow morphing within the SYmbol class hierarchy.
121// When they need to be used, Symbol*& is used to get the underlying object.
122// That lets us morph an object to another type during / parsing, while maintaining
123// the persistence of symbol uses in other objects.
124// A rule must be followed, which is to never store a pointer to a Symbol, either in a Symbol or another object.
125// Instead, store a reference to the pointer of, which can always be obtained using the member function stable_rp().
126// Actually, it's a reference to the second pair of the element for the symbol in the symbol table (the first element of the pair is the name)
127// The underlying collection std::unordered_map guarantees that references and pointers remain valid even if rehashing occurs.
