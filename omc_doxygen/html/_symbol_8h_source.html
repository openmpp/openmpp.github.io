88// Specialization of std::hash for omc::position.
89// Required for creating unordered_map for lookup of comment
90// text by start position.

92namespace std {
93 template <>
94 struct hash<omc::position>
95 {
96 size_t operator()(const omc::position& k) const
97 {
98 // Comments are uniquely determined by first position in location
99 return (
100 (hash<unsigned int>()(k.column))
101 ^ (hash<unsigned int>()(k.line) << 7) // assume max 128 columns
102 ^ (hash<unsigned int>()(*k.filename->cbegin()) << 10) // assume max 1024 lines
103 );
104 }
105 };
106}

122// When they need to be used, Symbol*& is used to get the underlying object.
123// That lets us morph an object to another type during / parsing, while maintaining
124// the persistence of symbol uses in other objects.
125// A rule must be followed, which is to never store a pointer to a Symbol, either in a Symbol or another object.
126// Instead, store a reference to the pointer of, which can always be obtained using the member function stable_rp().
127// Actually, it's a reference to the second pair of the element for the symbol in the symbol table (the first element of the pair is the name)
128// The underlying collection std::unordered_map guarantees that references and pointers remain valid even if rehashing occurs.
