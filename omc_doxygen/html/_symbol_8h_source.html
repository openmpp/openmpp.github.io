82// Specialization of std::hash for omc::position.
83// Required for creating unordered_map for lookup of comment
84// text by start position.

86namespace std {
87 template <>
88 struct hash<omc::position>
89 {
90 size_t operator()(const omc::position& k) const
91 {
92 // Comments are uniquely determined by first position in location
93 return (
94 (hash<unsigned int>()(k.column))
95 ^ (hash<unsigned int>()(k.line) << 7) // assume max 128 columns
96 ^ (hash<unsigned int>()(*k.filename->cbegin()) << 10) // assume max 1024 lines
97 );
98 }
99 };
100}

115// TODO: Explain better the use of double indirection to persist of Symbols and allow morphing within the SYmbol class hierarchy.
116// When they need to be used, Symbol*& is used to get the underlying object.
117// That lets us morph an object to another type during / parsing, while maintaining
118// the persistence of symbol uses in other objects.
119// A rule must be followed, which is to never store a pointer to a Symbol, either in a Symbol or another object.
120// Instead, store a reference to the pointer of, which can always be obtained using the member function stable_rp().
121// Actually, it's a reference to the second pair of the element for the symbol in the symbol table (the first element of the pair is the name)
122// The underlying collection std::unordered_map guarantees that references and pointers remain valid even if rehashing occurs.
