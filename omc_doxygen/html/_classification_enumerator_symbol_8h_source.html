37 string heuristic_short_name(void) const
38 {
39 string hn; // heuristic name
40
41 // Get the dimension's label for the model's default language.
42 string lbl = pp_labels[0];
43 string unm = unique_name;
44 if (lbl != unm && lbl.length() <= short_name_max_length) {
45 // Use label from model source code, if not too long.
46 hn = lbl;
47 }
48 else {
49 hn = unm;
50 }
51
52 assert(hn.length() > 0); // logic guarantee
53
54 // trim off leading "om_" prefix if present
55 //if (hn.starts_with("om_")) { // c++20
56 if (hn.length() >= 3 && hn.substr(0, 3) == "om_") {
57 hn.erase(0, 3);
58 }
59
60 if (!std::isalpha(hn[0], locale::classic())) {
61 // First character is not alphabetic.
62 // Prepend X_ to make valid name
63 hn = "X_" + hn;
64 }
65
66 // trim off trailing "_" if present
67 //if (hn.ends_with("_")) { // c++20
68 if (hn[hn.length() - 1] == '_') {
69 hn.erase(hn.length() - 1, 1);
70 }
71
72 // if name is subject to truncation, remove characters from middle rather than truncating from end,
73 // because long descriptions often disambiguate at both beginning and end.
74 if (hn.length() > short_name_max_length) {
75 // replacement string for snipped section
76 string r = "_x_";
77 // number of characters to snip from middle (excess plus length of replacement)
78 size_t n = hn.length() - short_name_max_length + r.length();
79 // start of snipped section (middle section of original string)
80 size_t p = (hn.length() - n) / 2;
81 // replace snipped section with _
82 if (p > 0 && p < hn.length()) {
83 hn.replace(p, n, r);
84 }
85 }
86
87 // Make name alphanumeric and truncate it to maximum length.
88 hn = openm::toAlphaNumeric(hn, short_name_max_length);
89
90 // trim off trailing "_" if present
91 //if (hn.ends_with("_")) { // c++20
92 if (hn[hn.length() - 1] == '_') {
93 hn.erase(hn.length() - 1, 1);
94 }
95
96 return hn;
97 }

103
104 // Perform post-parse operations specific to this level in the Symbol hierarchy.
