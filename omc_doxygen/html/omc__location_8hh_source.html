320  operator<< (std::basic_ostream<YYChar>& ostr, const location& loc)
321  {
322  location::counter_type end_col

324  ostr << loc.begin;

329  else if (loc.begin.line < loc.end.line)

331  else if (loc.begin.column < end_col)
