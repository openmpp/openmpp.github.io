198 string declaration(bool full = false, bool ellipses = false) const
199 {
200 string result;
201 if (statistic == token::TK_unit) {
202 result = "unit";
203 }
204 else {
205 if (ellipses) {
206 result = "...";
207 }
208 else {
209 auto a = pp_attribute;
210 assert(a); // all accumulators except unit have an attribute
211 result = a->pretty_name();
212 }
213 // surround attribute with table operator, increment, statistic if not defaults
214 if (full || (tabop != token::TK_interval)) {
215 result = token_to_string(tabop) + "(" + result + ")";
216 }
217 if (full || (increment != token::TK_delta)) {
218 result = token_to_string(increment) + "(" + result + ")";
219 }
220 if (full || (statistic != token::TK_sum)) {
221 result = token_to_string(statistic) + "(" + result + ")";
222 }
223 }
